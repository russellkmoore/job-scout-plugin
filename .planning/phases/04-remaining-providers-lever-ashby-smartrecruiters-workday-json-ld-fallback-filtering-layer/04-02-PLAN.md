From scripts/ats/providers/base.py (Provider Protocol — duck-typed, no inheritance):
```python
class Provider(Protocol):
    NAME: str  # e.g. "lever"
    BOARD_URL_PATTERNS: List[str]
    def detect(self, company_slug: str, name: str, client: "httpx.Client") -> DetectionResult: ...
    def board_url_from_url(self, url: str) -> Optional[str]: ...
    def fetch(self, slug: str, client: "httpx.Client", semaphore) -> FetchResult: ...
    def to_listing(self, payload: Dict[str, Any]) -> "Listing": ...

@dataclass(frozen=True)
class DetectionResult:
    provider: str
    status: DetectionStatus  # CONFIRMED | BORDERLINE | NOT_FOUND | ERROR
    board_url: Optional[str]
    confidence: float
    evidence: Dict[str, Any] = field(default_factory=dict)

@dataclass(frozen=True)
class FetchResult:
    provider: str
    company_slug: str
    listings: List["Listing"]
    raw: List[Dict[str, Any]]
    http_status: int
    auth_required: bool = False  # NEW from Plan 04-01 — Lever/Ashby ignore this (always False)
```
